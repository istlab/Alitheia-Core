/**
 * Hints to the metric activator on the ordering of the resource
 * version to be used.
 */
public enum InvocationOrder {
    /**Old entries first*/
    OLDFIRST,

    /**New entries first*/
    NEWFIRST,

    /**Randomized ordering*/
    RANDOM
}
